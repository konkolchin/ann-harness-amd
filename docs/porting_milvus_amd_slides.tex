%----- 4 -----
\begin{frame}[fragile]{How vector search works (end to end)}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Procedure}
      \begin{enumerate}
        \item \textbf{Store} many vectors in the database
              (here: 1M image descriptors).
        \item \textbf{Query} arrives: one new vector
              (``find items like this'').
        \item Engine finds the \textbf{k nearest}
              stored vectors (we use $k=10$).
        \item Returns their IDs (and distances)
              to the application.
      \end{enumerate}
      \vspace{0.3em}
      {\small Exact scan of 1M vectors is slow;
      IVF + GPU approximates this fast.}
    \end{column}
    \begin{column}{0.50\textwidth}
      \centering

%----- 5c -----
\begin{frame}{Next index: GPU\_IVF\_PQ (why)}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{IVF\_FLAT} (what we led with)
      \begin{itemize}
        \item Full float vectors in each list
        \item Best apples-to-apples CPU$\leftrightarrow$GPU story
        \item Higher memory
      \end{itemize}
      \vspace{0.6em}
      \textbf{IVF\_PQ} (product path)
      \begin{itemize}
        \item Same IVF lists, but vectors are
              \textbf{product-quantized} (compressed)
        \item Trade: lower recall, much less memory
        \item Knob \texttt{m} = number of PQ subspaces
              (SIFT dim 128 $\Rightarrow$ \texttt{m}$\in\{8,16,32,\ldots\}$)
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{block}{Lab recipe (AMD, 2026-07-21)}
        \footnotesize
        Host: RX 7900 XTX (\texttt{gfx1100})\\
        Index: \texttt{GPU\_IVF\_PQ}\\
        \texttt{nlist}=1024, \texttt{nbits}=8, \texttt{k}=10\\
        \texttt{m} $= 8,16,32$\\
        Batched harness (10k queries / search)\\
        Sealed path: \texttt{GPU\_CUVS\_IVF\_PQ}
      \end{block}
      \vspace{0.4em}
      {\small Done: same recipe on NVIDIA RTX 4080
      (see HIP vs CUDA slides).}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}[fragile]{What is recall@10?}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{In one sentence}
      \begin{itemize}
        \item For each query we know the
              \emph{true} 10 nearest neighbors
              (from a trusted ground truth).
        \item The system also returns 10 IDs.
        \item \textbf{recall@10} = fraction of the
              true top-10 that appear in the
              system's top-10.
      \end{itemize}
      \vspace{0.35em}
      \textbf{Example:} true top-10 has 10 IDs;
      system finds 8 of them $\rightarrow$
      recall@10 $= 0.8$ (80\%).
      \vspace{0.35em}

      {\small $1.0$ = perfect; lower = more misses.\\
      Our GPU table goes about $0.38 \rightarrow 0.98$
      as search gets more thorough.}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \begin{tikzpicture}[
        font=\tiny,
        hit/.style={circle, draw=green!60!black, fill=green!25,
                    minimum size=0.55cm, thick},
        miss/.style={circle, draw=red!70!black, fill=red!15,
                     minimum size=0.55cm, thick},
        other/.style={circle, draw=gray!60, fill=gray!10,
                      minimum size=0.55cm},
      ]
        \node[font=\scriptsize\bfseries] at (0,2.35) {Ground truth top-10};
        \foreach \i/\lab in {0/A,1/B,2/C,3/D,4/E,5/F,6/G,7/H,8/I,9/J} {
          \node[hit] at (-2.25+0.5*\i,1.7) {\lab};
        }

\begin{frame}[fragile]{What is IVF?}
  \begin{columns}[T]
    \begin{column}{0.50\textwidth}
      \textbf{IVF = Inverted File} (Approximate Nearest Neighbor index)
      \begin{itemize}
        \item At \textbf{build} time, group the
              1M vectors into many clusters
              (\texttt{nlist}=1024 lists).
        \item Each list has a \emph{center}
              (centroid) and the vectors that
              belong to it.
        \item At \textbf{search} time, do \emph{not}
              compare the query to all 1M vectors.
        \item Compare to list centers first, then
              scan vectors only in a few lists
              (see \texttt{nprobe} next).
      \end{itemize}
      \vspace{0.3em}
      {\small Our indexes: CPU \texttt{IVF\_FLAT},
      GPU \texttt{GPU\_IVF\_FLAT} --- same idea;
      FLAT = store full vectors in each list.}
    \end{column}
    \begin{column}{0.48\textwidth}
      \centering
      \begin{tikzpicture}[

\begin{frame}[fragile]{What is nprobe? (IVF search knob)}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \textbf{Given an IVF index}
      \begin{itemize}
        \item The query first finds the nearest
              list centers.
        \item \textbf{nprobe} = how many of those
              lists to open and scan
              (we used 1, 4, 8, 16, 32).
        \item Rest of the database is skipped
              --- that is the speedup vs a full scan.
      \end{itemize}
      \vspace{0.4em}
      \textbf{Trade-off}
      \begin{itemize}
        \item Higher \texttt{nprobe}: better recall, more work (usually lower QPS).
        \item Lower \texttt{nprobe}: faster, may miss true neighbors.
      \end{itemize}
    \end{column}
    \begin{column}{0.46\textwidth}
      \centering
      \begin{tikzpicture}[
